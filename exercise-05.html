sd_pop <- function(data, vars){

  results <- data.frame(
    variable = character(),
    mean = numeric(),
    sd_pop = numeric())

  for(i in vars){

    variable <- data[[i]]
    average <- mean(variable, na.rm = TRUE)
    diff <- variable - average
    s.size <- length(variable)
    sd.popu <- sqrt(sum(diff^2) / s.size)
    results <- rbind(results,
                     data.frame(variable = i,
                                mean = average,
                                sd_pop = sd.popu))
  }
  return(results)
}

mean.sd.resumen <- function(data, vars){

  results <- data.frame(
    variable = character(),
    mean = numeric(),
    sd_ = numeric())

  for(i in vars){

    variable <- data[[i]]
    average <- mean(variable, na.rm = TRUE)
    st.dev <- sd(variable, na.rm = TRUE)
    se.zombie <- st.dev/sqrt(nrow(data))
    results <- rbind(results,
                     data.frame(variable = i,
                                mean_z = average,
                                sd_z = average,
                                se_z = se.zombie))
  }
  return(results)
}

ci.95.zombie <- function(data, variables){
 ci.results <- data.frame(
   variable = character(),
   lower = numeric(),
   upper=numeric()
 )
 
 for(i in variables){
   average <- data$mean_z[data$variable == i]
    st.devi <- data$se_z[data$variable == i]

   variable <- data[[i]]
   ci <-  average + qnorm(c(0.025, 0.975))*st.devi
   
   ci.results <- rbind(ci.results,
                        data.frame(variable = i,
                                   lower = ci[1],
                                   upper = ci[2]))
 }
return(ci.results)
}
